function solution(arr) {
  const { length } = arr;
  const order = Array.from({ length }, () => 1);

  for (let i = 0; i < length; i++) {
    for (let j = 0; j < length; j++) {
      if (arr[i] < arr[j]) {
        order[i] += 1;
      }
    }
  }

  return order;
}
